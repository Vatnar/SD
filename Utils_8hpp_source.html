58inline std::expected<void, std::string>
59SingleTimeCommand(const vk::Device& device, const vk::Queue& queue,
60 const vk::CommandPool& commandPool,
61 const std::function<void(const vk::CommandBuffer&)>& action) {
62 vk::CommandBufferAllocateInfo allocInfo(commandPool, vk::CommandBufferLevel::ePrimary, 1);
63
64 auto allocRes = device.allocateCommandBuffers(allocInfo);
65 if (allocRes.result != vk::Result::eSuccess) {
66 return std::unexpected("Failed to allocate command buffers: " + vk::to_string(allocRes.result));
67 }
68 vk::CommandBuffer cmdBuffer = allocRes.value.front();
69
70 vk::CommandBufferBeginInfo beginInfo(vk::CommandBufferUsageFlagBits::eOneTimeSubmit);
71 if (auto res = cmdBuffer.begin(beginInfo); res != vk::Result::eSuccess) {
72 device.freeCommandBuffers(commandPool, cmdBuffer);
73 return std::unexpected("Failed to begin cmdBuffer: " + vk::to_string(res));
74 }
75
76 action(cmdBuffer);
77
78 if (auto res = cmdBuffer.end(); res != vk::Result::eSuccess) {
79 device.freeCommandBuffers(commandPool, cmdBuffer);
80 return std::unexpected("Failed to end cmdbuffer: " + vk::to_string(res));
81 }
82
83 auto fenceRes = device.createFenceUnique({});
84 if (fenceRes.result != vk::Result::eSuccess) {
85 device.freeCommandBuffers(commandPool, cmdBuffer);
86 return std::unexpected("Failed to create fence: " + vk::to_string(fenceRes.result));
87 }
88 vk::UniqueFence fence = std::move(fenceRes.value);
89
90 vk::SubmitInfo submitInfo;
91 submitInfo.setCommandBuffers(cmdBuffer);
92 if (auto res = queue.submit(submitInfo, *fence); res != vk::Result::eSuccess) {
93 device.freeCommandBuffers(commandPool, cmdBuffer);
94 return std::unexpected("Failed to submit to queue: " + vk::to_string(res));
95 }
96
97 if (auto res = device.waitForFences(*fence, VK_TRUE, std::numeric_limits<uint64_t>::max());
98 res != vk::Result::eSuccess) {
99 device.freeCommandBuffers(commandPool, cmdBuffer);
100 return std::unexpected("Failed to wait for fence: " + vk::to_string(res));
101 }
102
103 device.freeCommandBuffers(commandPool, cmdBuffer);
104 return {};
105}

107inline uint32_t FindMemoryType(const vk::PhysicalDevice& physicalDevice, uint32_t typeFilter,
108 vk::MemoryPropertyFlags properties) {
109 vk::PhysicalDeviceMemoryProperties memProperties = physicalDevice.getMemoryProperties();
110
111 for (uint32_t i = 0; i < memProperties.memoryTypeCount; i++) {
112 if ((typeFilter & (1 << i)) &&
113 (memProperties.memoryTypes[i].propertyFlags & properties) == properties) {
114 return i;
115 }
116 }
117 Engine::Abort("failed to find suitable memory type!");
118}

177inline void TransitionImageLayout(const vk::CommandBuffer& cmdBuffer, const vk::Image& image,
178 vk::Format format, vk::ImageLayout oldLayout,
179 vk::ImageLayout newLayout) {
180 // TODO: Use vk::ImageMemoryBarrier2 for better synchronization (requires Vulkan 1.3 or extension)
181 vk::ImageMemoryBarrier barrier;
182 barrier.oldLayout = oldLayout;
183 barrier.newLayout = newLayout;
184 barrier.srcQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
185 barrier.dstQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
186 barrier.image = image;
187 barrier.subresourceRange.aspectMask = vk::ImageAspectFlagBits::eColor;
188 barrier.subresourceRange.baseMipLevel = 0;
189 barrier.subresourceRange.levelCount = 1;
190 barrier.subresourceRange.baseArrayLayer = 0;
191 barrier.subresourceRange.layerCount = 1;
192
193 vk::PipelineStageFlags sourceStage;
194 vk::PipelineStageFlags destinationStage;
195
196 if (oldLayout == vk::ImageLayout::eUndefined &&
197 newLayout == vk::ImageLayout::eTransferDstOptimal) {
198 barrier.srcAccessMask = {};
199 barrier.dstAccessMask = vk::AccessFlagBits::eTransferWrite;
200
201 sourceStage = vk::PipelineStageFlagBits::eTopOfPipe;
202 destinationStage = vk::PipelineStageFlagBits::eTransfer;
203 } else if (oldLayout == vk::ImageLayout::eTransferDstOptimal &&
204 newLayout == vk::ImageLayout::eShaderReadOnlyOptimal) {
205 barrier.srcAccessMask = vk::AccessFlagBits::eTransferWrite;
206 barrier.dstAccessMask = vk::AccessFlagBits::eShaderRead;
207
208 sourceStage = vk::PipelineStageFlagBits::eTransfer;
209 destinationStage = vk::PipelineStageFlagBits::eFragmentShader;
210 } else {
211 Engine::Abort("unsupported layout transition!");
212 }
213
214 cmdBuffer.pipelineBarrier(sourceStage, destinationStage, {}, nullptr, nullptr, barrier);
215}

227 const vk::PhysicalDevice& physicalDevice,
228 const vk::Queue& graphicsQueue,
229 const vk::CommandPool& commandPool,
230 const std::filesystem::path& filePath) {

232 stbi_uc* pixels =
233 stbi_load(filePath.c_str(), &texWidth, &texHeight, &texChannels, STBI_rgb_alpha);
234 vk::DeviceSize imageSize = texWidth * texHeight * 4;
235
236 if (!pixels) {
237 return std::unexpected("Failed to load texture image: " + filePath.string());
238 }
239
240 auto [stagingBuffer, stagingBufferMemory] = CreateBuffer(
241 device, physicalDevice, imageSize, vk::BufferUsageFlagBits::eTransferSrc,
242 vk::MemoryPropertyFlagBits::eHostVisible | vk::MemoryPropertyFlagBits::eHostCoherent);
243
244 void* data = CheckVulkanResVal(device.mapMemory(*stagingBufferMemory, 0, imageSize),
245 "Failed to map texture image: ");
246
247 memcpy(data, pixels, imageSize);
248 device.unmapMemory(*stagingBufferMemory);
249
250 stbi_image_free(pixels);
251
252 auto [image, imageMemory] =
253 CreateImage(device, physicalDevice, texWidth, texHeight, vk::Format::eR8G8B8A8Srgb,
254 vk::ImageTiling::eOptimal,
255 vk::ImageUsageFlagBits::eTransferDst | vk::ImageUsageFlagBits::eSampled,
256 vk::MemoryPropertyFlagBits::eDeviceLocal);
257
258 auto cmdRes = SingleTimeCommand(
259 device, graphicsQueue, commandPool, [&](const vk::CommandBuffer& cmdBuffer) {
260 TransitionImageLayout(cmdBuffer, *image, vk::Format::eR8G8B8A8Srgb,
261 vk::ImageLayout::eUndefined, vk::ImageLayout::eTransferDstOptimal);
262 CopyBufferToImage(cmdBuffer, *stagingBuffer, *image, static_cast<uint32_t>(texWidth),
263 static_cast<uint32_t>(texHeight));
264 TransitionImageLayout(cmdBuffer, *image, vk::Format::eR8G8B8A8Srgb,
265 vk::ImageLayout::eTransferDstOptimal,
266 vk::ImageLayout::eShaderReadOnlyOptimal);
267 });
268
269 if (!cmdRes) {
270 return std::unexpected(cmdRes.error());
271 }
272
273 vk::ImageViewCreateInfo viewInfo({}, *image, vk::ImageViewType::e2D, vk::Format::eR8G8B8A8Srgb,
274 {}, {vk::ImageAspectFlagBits::eColor, 0, 1, 0, 1});
275 vk::UniqueImageView imageView = CheckVulkanResVal(device.createImageViewUnique(viewInfo),
276 "Failed to create unique image view: ");
277
278 return Texture{std::move(image), std::move(imageMemory), std::move(imageView)};
279}
